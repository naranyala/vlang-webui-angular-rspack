1. **Added RateLimiter in security.v:**

```v
pub struct RateLimiter {
pub mut:
    requests      map[string][]u64
    max_requests  int
    window_seconds int
}

pub fn new_rate_limiter(max_requests int, window_seconds int) &RateLimiter {
    return &RateLimiter{
        requests: map[string][]u64{}
        max_requests: max_requests
        window_seconds: window_seconds
    }
}

pub fn (mut rl RateLimiter) is_allowed(key string) bool {
    now := u64(time.now().unix())
    window_start := now - u64(rl.window_seconds)

    requests := rl.requests[key] or { []u64{} }

    // Filter to only requests within the window
    mut valid_requests := []u64{}
    for req_time in requests {
        if req_time >= window_start {
            valid_requests << req_time
        }
    }

    // Check if under limit
    if valid_requests.len >= rl.max_requests {
        rl.requests[key] = valid_requests
        return false
    }

    // Add this request
    valid_requests << now
    rl.requests[key] = valid_requests
    return true
}
```

2. **Created RateLimitMiddleware:**
```v
pub struct RateLimitMiddleware {
pub mut:
    limiter          &security.RateLimiter
    global_limiter   &security.RateLimiter
    ip_limiter       &security.RateLimiter
    user_limiter     &security.RateLimiter
}

pub fn (mut m RateLimitMiddleware) check(ip string, user_id string) RateLimitResult {
    // Check global limit first
    if !m.global_limiter.is_allowed('global') {
        return RateLimitResult{
            allowed: false
            remaining: 0
            retry_after: 60
        }
    }

    // Check IP-based limit
    ip_remaining := m.ip_limiter.get_remaining_requests(ip)
    if !m.ip_limiter.is_allowed(ip) {
        return RateLimitResult{
            allowed: false
            remaining: 0
            retry_after: 60
        }
    }

    // Check user-based limit if authenticated
    if user_id != '' {
        if !m.user_limiter.is_allowed(user_id) {
            return RateLimitResult{
                allowed: false
                remaining: 0
                retry_after: 60
            }
        }
    }

    return RateLimitResult{
        allowed: true
        remaining: ip_remaining - 1
        retry_after: 0
    }
}
```
