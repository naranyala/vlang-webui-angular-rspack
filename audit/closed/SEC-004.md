### Changes Made

1. **Added input sanitization in security.v:**
```v
pub fn sanitize_input(input string) string {
    // Remove null bytes
    mut result := input.replace('\x00', '')
    
    // Remove/escape common injection characters
    result = result.replace('<', '&lt;')
    result = result.replace('>', '&gt;')
    result = result.replace('"', '&quot;')
    result = result.replace('\'', '&#x27;')
    
    return result.trim_space()
}
```

2. **Created ValidationMiddleware:**
```v
pub struct ValidationMiddleware {
pub mut:
    max_body_size    int
    max_string_len   int
    allowed_html_tags []string
}

pub fn (m ValidationMiddleware) validate(input map[string]string) ValidationResult {
    mut result := ValidationResult{
        is_valid: true
        errors: []string{}
        sanitized: map[string]string{}
    }

    for key, value in input {
        // Validate key
        if key.len > 100 {
            result.is_valid = false
            result.errors << 'Key too long'
            continue
        }

        // Validate value length
        if value.len > m.max_string_len {
            result.is_valid = false
            result.errors << 'Value too long'
            continue
        }

        // Sanitize value
        sanitized := security.sanitize_input(value)
        result.sanitized[key] = sanitized
    }

    return result
}
```
