2. **Implemented password hashing in security.v:**
```v
pub fn hash_password(password string) !string {
    // Generate random salt (16 bytes)
    mut salt_bytes := [16]u8{}
    crypto.rand.bytes(mut salt_bytes) or {
        return error('Failed to generate salt')
    }
    salt := hex.encode(salt_bytes)

    // Generate hash using password + salt with multiple iterations
    hash := pbkdf2_simple(password, salt, 10000)

    // Return format: $salt$hash
    return '\$${salt}\$${hash}'
}

pub fn verify_password(password string, hash_string string) bool {
    // Parse salt and hash from stored string
    parts := hash_string.split('\$')
    if parts.len != 3 {
        return false
    }

    salt := parts[1]
    stored_hash := parts[2]

    // Compute hash with same salt
    computed_hash := pbkdf2_simple(password, salt, 10000)

    // Constant-time comparison to prevent timing attacks
    return secure_compare(computed_hash, stored_hash)
}
```

3. **Updated AuthService to use secure password handling:**
```v
pub fn (s &AuthService) register_user(username string, password string, email string) bool {
    if s.users.exists(username) {
        return false
    }

    // SEC-001: Validate password strength
    security.validate_password_strength(password) or {
        return false  // Password too weak
    }

    // SEC-001: Hash password before storage (never store plaintext)
    password_hash := security.hash_password(password) or {
        return false  // Hashing failed
    }

    now := u64(time.now().unix())
    s.users[username] = UserInfo{
        id: 'user_${now}'
        username: username
        email: email
        password_hash: password_hash  // SEC-001: Store hash, not plaintext
        roles: ['user']
        permissions: ['read']
        created_at: now
        last_login: 0
    }

    return true
}

pub fn (s &AuthService) authenticate(username string, password string) ?AuthResult {
    user := s.users[username] or {
        return error('User not found')
    }

    // SEC-001: Verify password against stored hash (not plaintext comparison)
    if !security.verify_password(password, user.password_hash) {
        return error('Invalid credentials')
    }

    // ... rest of authentication
}
```
